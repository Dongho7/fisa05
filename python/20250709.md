like this...

```python
# 은행계좌를 만드는 BankAccount라는 클래스를 만들거에요.

# - 인스턴스 변수(self): 계좌번호(__account), 이름(__name), 통장잔고(__balance), 비밀번호(__password)
# getter(조회), setter(삽입, 변경)
# - 인스턴스 메서드(self):
                    # - 금액을 입금하는 메서드(put_balance),
                    # - 금액 출금하는 메서드(out_balance)
                    # - 금액 조회(get_balance),
                    # - 비밀번호 변경(set_pw)
                    # - 비밀번호 조회(get_pw)
# - 클래스 변수: 현재까지 몇 명이 우리 은행에서 계좌를 만들었는지(customer_total)
# - 클래스 메서드: 현재까지 몇 명이 우리 은행에서 계좌를 만들었는지 확인(customer_total_check)


class BankAccount:
    customer_total = 0

    @classmethod 
    def customer_total_check(cls):
        print(f"현재까지 우리 은행에 계좌를 만든 사람은 {cls.customer_total}명입니다.")

    def __init__(self, __account, __name, __password, __balance=0):
        self.__account = __account
        self.__name = __name
        self.__balance = __balance
        self.__password = __password
    
        print(f"{__account} 계좌번호로 {__name}님의 계좌가 만들어졌습니다. 현재 통장잔고는 {__balance}원이며, 비밀번호는 {__password}입니다. ")
        type(self).customer_total += 1

    def put_balance(self, money:int):
        print(f"{money}원 입금되었습니다.")
        self.__balance += money
    
    def out_balance(self, money:int):
        if self.__balance >= money:
            self.__balance -= money
            print(f"{money}원 출금되었습니다.")
        else:
            print(f"잔액 부족. 현재 잔고는 {self.__balance}원입니다.")
    
    
    @property
    def get_balance(self):
        print(f"현재 잔고 {self.__balance}원")

    def set_pw(self, new_password):
        self.__password = new_password
        print(f"{new_password}로 패스워드가 변경되었습니다")

    def get_pw(self):
        print(f"{self.__password}가 현재 패스워드입니다.")
```

## Tax service 
```python
class TaxService:
    def __init__(self, tax_rate=0.15):
        self.tax_rate = tax_rate

    def calculate_tax(self, amount):
        tax = amount * self.tax_rate
        print(f"세금은 {tax:.0f}원입니다.")
        return tax

    def issue_tax_report(self):
        print("세금 신고서를 발급합니다.")
```
